Code Examples

Learn Defacto by example. These samples demonstrate various language features.

Hello World

#Mainprogramm.start #NO_RUNTIME #SAFE <.de var msg: string = "Hello, World!" static.pl> display{msg} .> #Mainprogramm.end

Compile: defacto -terminal hello.de

Variables and Types

#Mainprogramm.start #NO_RUNTIME #SAFE <.de # Integer types var count: i32 = 42 var big: i64 = 1000000 var byte: u8 = 255 # String var msg: string = "Hello" # Array var arr: i32[10] # Constant const MAX: i32 = 100 static.pl> display{msg} .> #Mainprogramm.end

Arithmetic Operations

#Mainprogramm.start #NO_RUNTIME #SAFE <.de var a: i32 = 10 var b: i32 = 5 var result: i32 = 0 static.pl> # Addition result = (a + b) # Subtraction result = (a - b) # Multiplication result = (a * b) # Division result = (a / b) .> #Mainprogramm.end

Structures

#Mainprogramm.start #NO_RUNTIME #SAFE # Define a structure struct Player { x: i32 y: i32 health: i32 score: i32 } <.de var player: Player static.pl> # Initialize fields player.x = 100 player.y = 200 player.health = 100 player.score = 0 # Modify fields player.health = (player.health - 10) player.score = (player.score + 100) .> #Mainprogramm.end

If/Else Statements

#Mainprogramm.start #NO_RUNTIME #SAFE <.de var x: i32 = 5 var msg: string = "equal" var err: string = "not equal" static.pl> # Simple if if x == 5 { display{msg} } # If/else if x == 10 { display{msg} } else { display{err} } .> #Mainprogramm.end

Loop

#Mainprogramm.start #NO_RUNTIME #SAFE <.de var counter: i32 = 10 static.pl> loop { # Exit condition if counter == 0 { stop } # Decrement counter = (counter - 1) } .> #Mainprogramm.end

Functions

#Mainprogramm.start #NO_RUNTIME #SAFE # Function definition function == print_hello { <.de var msg: string = "Hello from function!" static.pl> display{msg} .> } function == add_numbers { <.de var a: i32 = 10 var b: i32 = 5 var result: i32 = 0 static.pl> result = (a + b) .> } <.de static.pl> # Call functions call #print_hello call #add_numbers .> #Mainprogramm.end

Keyboard Driver

#Mainprogramm.start #NO_RUNTIME #SAFE # Define keyboard driver <drv. Const.driver = keyboard_driver keyboard_driver <<func = keyboard>> .dr> <.de var key: i32 = 0 var msg: string = "Press a key" static.pl> display{msg} # Call the driver call #keyboard_driver # Read the key readkey{key} display{key} .> #Mainprogramm.end

Graphics (Bare-Metal)

#Mainprogramm.start #NO_RUNTIME #SAFE <.de var msg: string = "Colored Text" static.pl> # Set green color color{10} display{msg} # Clear screen clear{} .> #Mainprogramm.end

Complete Program

#Mainprogramm.start #NO_RUNTIME #SAFE # Structure definition struct Point { x: i32 y: i32 } # Function definition function == print_point { <.de var msg: string = "Point printed" static.pl> display{msg} .> } # Main code section <.de var p: Point var msg: string = "Hello, Defacto!" static.pl> # Initialize struct p.x = 10 p.y = 20 # Display message display{msg} # Call function call #print_point .> #Mainprogramm.end
